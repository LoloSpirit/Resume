\newcommand{\roundpic}[2][1.5cm]{%
  \begin{tikzpicture}[remember picture,overlay]
    % Background circle
    \fill[gray!20] (current page.north east) ++(-2.8cm,-2.4cm) circle (#1);
    \clip (current page.north east) ++(-2.8cm,-2.4cm) circle (#1);
    \node at ($(current page.north east)+(-2.8cm,-2.4cm)$) {\includegraphics[width=1.7\dimexpr#1]{#2}};
    \draw[thick, gray] ($(current page.north east)+(-2.8cm,-2.4cm)$) circle (#1);
  \end{tikzpicture}
}

    \end{tabular*}
  }
